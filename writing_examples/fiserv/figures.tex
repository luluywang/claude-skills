\caption{Illustration of Payment Flows\label{fig:payment-flow}}
\small

\[
\xymatrix{ & *+++++[F:<5pt>]{\text{Network}}\\*
++++[F:<5pt>]{\text{Acquirer}}\ar@(u,l)[ru]_{0.07}^{\text{Assessment\ }}\ar[rr]_{1.90}^{\text{Interchange}} & & *+++++[F:<5pt>]{\text{Issuer}}\ar@(u,r)[lu]_{\text{\ Assessment}}^{0.07}\ar[d]_{\text{Rewards}}^{1.40}\\* +++++[F:<5pt>]{\text{Merchant}}\ar[u]_{2.25}^{\text{Merchant Discount}} & & *+++++[F:<5pt>]{\text{Consumer}}\ar[ll]_{\text{Purchase}}^{100} }
\]

\captionnotes{\textit{Notes:} Figure \ref{fig:payment-flow} displays average fee levels for a credit card transaction in the U.S., showing how fees flow through the payment system. Merchant fees come from \citet{Nilson2023}. Interchange fees and rewards are calculated from regulatory data on large banks \citep{Drechsler.etal2025}. Network fees are obtained from acquirer disclosures and Visa's financial statements \citep{Visa2020,Helcim2021}.}

\end{figure}
